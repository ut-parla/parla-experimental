430  def _process_operations(self, operations: List[MemoryOperation], slices: SlicesType = None) -> None:
431  """
432  Process the given memory operations.
433  Data will be moved, and protocol states is kept unchanged.
434  """
435  for op in operations:
436  if op.inst == MemoryOperation.NOOP:
437  pass # do nothing
438  elif op.inst == MemoryOperation.LOAD:
439  if MemoryOperation.LOAD_SUBARRAY in op.flag:
440  # build slices mapping first
441  self._array_array.set_slices_mapping(op.dst, slices)
442 
443  # check flag to see if dst is current device
444  dst_is_current_device = op.flag != MemoryOperation.SWITCH_DEVICE_FLAG
445 
446  # copy data
447  self._array_array.copy_data_between_device(
448  op.dst, op.src, dst_is_current_device)
449 
450  # sync stream before set it as ready, so asyc call is ensured to be done
451  if num_gpu > 0:
452  cupy.cuda.stream.get_current_stream().synchronize()
453  elif op.inst == MemoryOperation.EVICT:
454  scheduler = get_scheduler()
455  src_global_dev_id = scheduler.device_manager.parrayid_to_globalid(
456  op.src)
457  if self._cy_parray_cy_parray.get_num_active_tasks(src_global_dev_id) == 0:
458  # If none of visible tasks will refer this PArray, release
459  # this PArray instance of the source device from the PArray tracker.
460  scheduler.release_parray(
461  self._cy_parray_cy_parray, src_global_dev_id)
462  # decrement the reference counter, relying on GC to free the memor
463  self._array_array.clear(op.src)
464  elif op.inst == MemoryOperation.ERROR:
465  raise RuntimeError(
466  "PArray gets an error from coherence protocol")
467  else:
468  raise RuntimeError(f"PArray gets invalid memory operation from coherence protocol, "
469  f"detail: opcode {op.inst}, dst {op.dst}, src {op.src}")
