266 void InnerScheduler::task_cleanup_postsync(InnerWorker *worker, InnerTask *task,
267  int state) {
268  NVTX_RANGE("Scheduler::task_cleanup_postsync", NVTX_COLOR_MAGENTA)
269 
270  // std::cout << "Task Cleanup Post Sync" << std::endl;
271 
272  if (state == Task::RUNAHEAD) {
273  this->decrease_num_active_tasks();
274  task->notify_dependents_completed();
275  }
276 
277  // Release all resources for this task on all devices
278  for (size_t i = 0; i < task->assigned_devices.size(); ++i) {
279  Device *device = task->assigned_devices[i];
280  DevID_t dev_id = device->get_global_id();
281  ResourcePool_t &device_pool = device->get_reserved_pool();
282  ResourcePool_t &task_pool =
283  task->device_constraints[device->get_global_id()];
284 
285  // TODO(wlr): This needs to be changed to not release PARRAY resources
286  device_pool.increase<ResourceCategory::All>(task_pool);
287 
288  // PArrays could be evicted even during task barrier continuation.
289  // However, these PArrays will be allocated and tracked
290  // again after the task restarts.
291  if (!task->is_data_task()) {
292  for (size_t j = 0; j < task->parray_list[i].size(); ++j) {
293  parray::InnerPArray *parray = task->parray_list[i][j].first;
294  parray->decr_num_active_tasks(dev_id);
295  // This PArray is not released from the PArray tracker here,
296  // but when it is EVICTED, it will check the number of referneces
297  // and will be released if that is 0.
298  }
299  }
300  }
301 
302  // Clear all assigned streams from the task
303  task->streams.clear();
304  this->launcher->num_running_tasks--;
305  worker->remove_task();
306 
307  if (state == Task::RUNNING) {
308  task->reset();
309  auto status = task->process_dependencies();
310  this->enqueue_task(task, status);
311  }
312 
313  this->enqueue_worker(worker);
314 }
