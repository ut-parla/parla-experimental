192 TaskState InnerTask::add_dependent_task(InnerTask *task) {
193 
194  // Store all dependents for bookkeeping
195  // Dependents can be written to by multiple threads calling this function
196  // Dependents is read when the task is in cleanup, which can overlap with this
197  // function This write needs to be thread-safe
198 
199  // NOTE: This is not a lock free implementation. I don't know how to make it
200  // lock free
201  // This lock and its match in notify_dependents ensures correctness,
202  // explained above. A tasks "completeness" cannot change while we are
203  // adding a dependent to the list This means notify dependents has not
204  // run yet.
205  this->dependents.lock();
206 
207  TaskState state = this->get_state(); // s1
208  this->dependents.push_back_unsafe(task); // s3
209 
210  this->dependents.unlock();
211 
212  return state;

293 TaskStatusFlags InnerTask::notify(TaskState dependency_state, bool is_data) {
294 
295  bool spawnable = false;
296  bool mappable = false;
297  bool reservable = false;
298  bool compute_runnable = false;
299  bool runnable = false;
300 
301  if (is_data) {
302  if (dependency_state == TaskState::RUNAHEAD) {
303  // A data task never notifies for the other stages
304  runnable = (this->num_blocking_dependencies.fetch_sub(1) == 1);
305  }
306  } else {
307  if (dependency_state == TaskState::RUNAHEAD) {
308  compute_runnable =
309  (this->num_blocking_compute_dependencies.fetch_sub(1) == 1);
310  runnable = (this->num_blocking_dependencies.fetch_sub(1) == 1);
311  } else if (dependency_state >= TaskState::RESERVED) {
312  reservable = (this->num_unreserved_dependencies.fetch_sub(1) == 1);
313  } else if (dependency_state >= TaskState::MAPPED) {
314  mappable = (this->num_unmapped_dependencies.fetch_sub(1) == 1);
315  } else if (dependency_state >= TaskState::SPAWNED) {
316  spawnable = (this->num_unspawned_dependencies.fetch_sub(1) == 1);
317  }
318  }
319 
320  TaskStatusFlags status;
321  status.spawnable = spawnable;
322  status.mappable = mappable;
323  status.reservable = reservable;
324  status.compute_runnable = compute_runnable;
325  status.runnable = runnable;
326 
327  return status;
328 }
