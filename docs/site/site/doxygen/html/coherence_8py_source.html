451  def evict(self, device_id: int, keep_one_copy: bool = True) -> List[MemoryOperation]:
452  """ Tell the protocol that this device want to clear the copy.
453  Args:
454  device_id: id of this device
455  keep_one_copy: if true, writeback the last copy to CPU
456  Return:
457  List[MemoryOperation], could return several MemoryOperations.
458  And the order operations matter.
459  Or [ERROR] if this device has the last copy and `keep_one_copy` is false,
460  since eviction cannot be performed.
461  """
462  device_local_state = self._local_states_local_states[device_id]
463  operations = []
464  evict_last_copy = False
465 
466  if device_local_state == self.INVALIDINVALID: # already evicted, do nothing
467  operations.append(MemoryOperation.noop())
468  elif device_local_state == self.SHAREDSHARED:
469  if device_id == self.ownerowner: # has a chance this is the last copy
470  # find new owner
471  new_owner = None
472  for device, state in self._local_states_local_states.items():
473  if state == self.SHAREDSHARED and device != device_id: # should not include this device itself
474  new_owner = device
475  break
476 
477  # this device owns the last copy
478  if new_owner is None:
479  evict_last_copy = True
480  else:
481  self.ownerowner = new_owner
482  else: # Modified, this device owns the last copy
483  evict_last_copy = True
484 
485  if evict_last_copy:
486  if keep_one_copy: # write back to CPU
487  if device_id != CPU_INDEX:
488  operations.extend(self._write_back_to_write_back_to(CPU_INDEX, self.MODIFIEDMODIFIED, on_different_device=True, this_device_id=device_id))
489  # special case, since `this_device_id` is set, _write_back will not evict this devic
490  # need to do it manually

494  self._local_states_local_states[CPU_INDEX] = self.MODIFIEDMODIFIED
495  self._is_complete_is_complete[device_id] = True
