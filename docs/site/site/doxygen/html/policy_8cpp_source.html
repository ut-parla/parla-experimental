101  // If any device was chosen as a candidate device,
102  // it is set to True and the best_score is used to be compared
103  // and find the next better candidate device.
104  bool any_device_chosen{false};
105  std::shared_ptr<DeviceRequirement> best_device_req{nullptr};
106  // std::cout << task->get_name() << "inside arch req mapping. " << std::endl;
107  // TODO(wlr): Is this unused??
108  uint32_t i{0};
109  bool is_arch_available{false};
110  // For now the architecture placement has one resource requirement
111  // regardless of the devices of the architecture. In the future,
112  // we will allow a separate placement for each device.
113  auto placement_options = arch_placement_req->GetDeviceRequirementOptions();
114  int n_options = placement_options.size();
115  int start_idx = ++this->rrcount;
116 
117  for (int k = 0; k < n_options; k++) {
118  int idx = (start_idx + k) % n_options;
119  std::shared_ptr<DeviceRequirement> dev_req = placement_options[idx];
120  Score_t score{0};
121  DevID_t dev_global_id = dev_req->device()->get_global_id();
122  if (is_dev_assigned != nullptr &&
123  (*is_dev_assigned)[dev_global_id] == true) {
124  // If this architecture placement is the member of a
125  // multi-device task and this visiting device is already chosen
126  // as one of the placements, skip it.
127  continue;
128  }
129  bool is_dev_available = this->calc_score_devplacement(task, dev_req, mapper,
130  &score, parray_list);
131  if (!is_dev_available) {
132  continue;
133  }
134  is_arch_available = true;
135  if (!any_device_chosen || best_score <= score) {
136  best_score = score;
137  best_device_req = dev_req;
138  any_device_chosen = true;
139  }
140  ++i;
141  }
142  assert(best_device_req != nullptr);
143  chosen_dev_req = best_device_req;

148 bool LocalityLoadBalancingMappingPolicy::calc_score_mdevplacement(
149  InnerTask *task, MultiDeviceRequirements *mdev_placement_req,

156  const std::vector<std::shared_ptr<SinglePlacementRequirementBase>>
157  &placement_reqs_vec = mdev_placement_req->get_placement_reqs_ref();
158  member_device_reqs->resize(placement_reqs_vec.size());
159  // Task mapper does not allow to map a multi-device task to the same device
160  // multiple times. This vector marks an assigned device and filter it

162  std::vector<bool> is_dev_assigned(
163  this->device_manager_->get_num_devices<DeviceType::All>(), false);
164  // Iterate requirements of the devices specified in multi-device placement.
165  // All of the member devices should be available.
166  for (DevID_t did = 0; did < placement_reqs_vec.size(); ++did) {
167  std::shared_ptr<SinglePlacementRequirementBase> placement_req =

170  Score_t score{0};
171  bool is_member_device_available{false};
172  if (placement_req->is_dev_req()) {
173  dev_req = std::dynamic_pointer_cast<DeviceRequirement>(placement_req);
174  DevID_t dev_global_id = dev_req->device()->get_global_id();
175  if (!is_dev_assigned[dev_global_id]) {
176  is_member_device_available = this->calc_score_devplacement(
177  task, dev_req, mapper, &score, parray_list[did]);
178  if (is_member_device_available) {
179  is_dev_assigned[dev_global_id] = true;
180  }
181  }
182  } else if (placement_req->is_arch_req()) {
183  ArchitectureRequirement *arch_req =
184  dynamic_cast<ArchitectureRequirement *>(placement_req.get());
185  is_member_device_available = this->calc_score_archplacement(
186  task, arch_req, mapper, dev_req, &score, parray_list[did],
187  &is_dev_assigned);
188  if (is_member_device_available) {
189  DevID_t dev_global_id = dev_req->device()->get_global_id();
190  is_dev_assigned[dev_global_id] = true;
191  }
192  }
193  assert(dev_req != nullptr);
194  (*member_device_reqs)[did] = dev_req;
195  *average_score += score;
196  if (!is_member_device_available) {
197  // If any of the device candidates is not available,
198  // return false and exclude this option from task mapping.
199  return false;
200  }
201  }
202  *average_score /= placement_reqs_vec.size();
203  return true;

206 void LocalityLoadBalancingMappingPolicy::run_task_mapping(
207  InnerTask *task, const Mapper &mapper,

210  &parray_list,
211  std::vector<std::shared_ptr<PlacementRequirementBase>> *placement_req_options_vec) {
212  // A set of chosen devices to a task.
213  Score_t best_score{-1};
214  // If any device was chosen as a candidate device,
215  // it is set to True and the best_score is used to be compared
216  // and find the next better candidate device.
217  bool any_device_chosen{false};
218 
219  // Iterate all placement requirements passed by users and calculate
220  // scores based on a policy.
221  for (std::shared_ptr<PlacementRequirementBase> base_req :
222  *placement_req_options_vec) {
223  if (base_req->is_multidev_req()) {
224  // Multi-device placement requirements.
225  // std::cout << "[Multi-device requirement]\n";
226  MultiDeviceRequirements *mdev_reqs =
227  dynamic_cast<MultiDeviceRequirements *>(base_req.get());
228  std::vector<std::shared_ptr<DeviceRequirement>> mdev_reqs_vec;
229  Score_t score{0};
230  bool is_req_available = calc_score_mdevplacement(
231  task, mdev_reqs, mapper, &mdev_reqs_vec, &score, parray_list);
232  if (!is_req_available) {
233  continue;
234  }
235  if (best_score <= score || !any_device_chosen) {
236  best_score = score;
237  chosen_devices->swap(mdev_reqs_vec);
238  any_device_chosen = true;
239  }
240  } else if (base_req->is_dev_req()) {
241  // A single device placement requirement.
242  std::shared_ptr<DeviceRequirement> dev_req =
243  std::dynamic_pointer_cast<DeviceRequirement>(base_req);
244  Score_t score{0};
245  bool is_req_available = calc_score_devplacement(
246  task, dev_req, mapper, &score, parray_list[0]);
247  if (!is_req_available) {
248  continue;

250  if (best_score <= score || !any_device_chosen) {
251  assert(dev_req != nullptr);
252  best_score = score;
253  chosen_devices->clear();
254  chosen_devices->emplace_back(dev_req);
255  any_device_chosen = true;
256  }
257  } else if (base_req->is_arch_req()) {
258  // A single architecture placement requirement.
259  ArchitectureRequirement *arch_req =
260  dynamic_cast<ArchitectureRequirement *>(base_req.get());
261  std::shared_ptr<DeviceRequirement> chosen_dev_req{nullptr};
262  Score_t chosen_dev_score{0};

268  parray_list[0]);
269  if (!is_req_available) {
270  continue;
271  }
272  if (best_score <= chosen_dev_score || !any_device_chosen) {
273  assert(chosen_dev_req != nullptr);
