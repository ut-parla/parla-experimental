481  def get_slices_hash(self, global_slices: SlicesType) -> int:
482  """
483  Get hash value of a slices of complete array.
484 
485  This could be done by replaing list and slice to tuple
486  """
487  # little chance to have collision, but what if it happened?
488  hash_value = 17 # use a none zero hash value, so hash(0) != 0
489  prime = 31
490  if not isinstance(global_slices, tuple):
491  if isinstance(global_slices, list):
492  # Built-int hash() method might return negtive value.
493  # c_size_t is to ensure it is not negative
494  hash_value = hash_value * prime + ctypes.c_size_t(hash(tuple(global_slices))).value
495  elif isinstance(global_slices, slice):
496  hash_value = hash_value * prime + ctypes.c_size_t(hash(global_slices.indices(self.shapeshape[0]))).value
497  else:
498  hash_value = hash_value * prime + ctypes.c_size_t(hash(global_slices)).value
499  else:
500  if len(self.shapeshape) < len(global_slices):
501  raise IndexError(f"index out of range, index:{global_slices}")
502 
503  for d in range(len(global_slices)):
504  index = global_slices[d]
505  if isinstance(index, list):
506  hash_value = hash_value * prime + ctypes.c_size_t(hash(tuple(index))).value
507  elif isinstance(index, slice):
508  hash_value = hash_value * prime + ctypes.c_size_t(hash(index.indices(self.shapeshape[d]))).value
509  else:
510  hash_value = hash_value * prime + ctypes.c_size_t(hash(index)).value
