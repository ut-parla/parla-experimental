858  num_bulk_tasks_last_level = (
859  num_bulk_tasks % num_levels_for_bulk_tasks) + num_bulk_tasks_per_level
860  # Calculate the number of tasks per gpu per level.
861  num_bulk_tasks_per_gpu = (num_bulk_tasks_per_level) // num_gpus
862  """
863  for l in range(levels + 1):
864  if l % 2 > 0:
865  print(f"Level {l}: -- 1 --", flush=True)
866  else:
867  if l == levels:
868  print(f"Level {l}: {num_bulk_tasks_last_level}, {num_bulk_tasks_per_gpu}", flush=True)
869  else:

882  single_data_block_size = tgraph_config.total_data_width

884  if d > 0:

895  assert len(task_config.configurations) == 1

905  # tasks.
